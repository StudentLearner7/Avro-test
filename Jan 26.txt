import java.util.ArrayList;
import java.util.List;

public class ListPagination {
    public static void main(String[] args) {
        // Create a sample List<Number> with 10,000 elements
        List<Number> numberList = new ArrayList<>();
        for (int i = 1; i <= 10000; i++) {
            numberList.add(i);
        }

        // User input for page and size
        int page = 1; // Example: User input for page
        int size = 20; // Example: User input for size

        // Calculate the start and end indices for the requested page and size
        int startIndex = page * size;
        int endIndex = Math.min(startIndex + size, numberList.size());

        // Check if the requested page is within bounds
        if (startIndex < numberList.size()) {
            // Extract the sublist for the requested page
            List<Number> pageData = numberList.subList(startIndex, endIndex);

            // Print the retrieved data for the page
            for (Number num : pageData) {
                System.out.print(num + " ");
            }
        } else {
            System.out.println("Page is out of bounds.");
        }
    }
}
